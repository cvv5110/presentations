                \end{figure}}
        \end{column}
        \hspace*{-0.2in}
        \begin{column}{0.4\textwidth}
            \visible<4->{
            \begin{graybox}{Parametrization}
                \visible<5->{
                Parameters,
                \vspace*{-0.05in}
                \begin{equation*}
                    \vp = \left(\vOmega_0,\vOmega_f,\vxi\right)^\top\in\mathbb{R}^{n_p}
                \end{equation*}}
                \visible<6->{
                are, e.g., normally distributed,
                \vspace*{-0.05in}
                \begin{equation*}
                    \vp\sim\mathcal{N}\left(\mathbf{m},\boldsymbol{\sigma}\right)
                \end{equation*}}
                \visible<7->{
                where,
                \vspace*{-0.05in}
                \begin{itemize}
                    \item $\mathbf{m}$: nominal mission,
                    \item $\boldsymbol{\sigma}$: mission uncertainty
                \end{itemize}}
            \end{graybox}}
        \end{column}
    \end{columns}
\end{frame}
